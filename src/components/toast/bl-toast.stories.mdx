import { html } from 'lit';
import { ifDefined } from 'lit/directives/if-defined.js';

import { Meta, Canvas, Story } from '@storybook/addon-docs';

<Meta

    },

    }

export const ToastTemplate = (args) => html`
<bl-toast
    description='${ifDefined(args.description)}'
    variant='${ifDefined(args.variant)}'
    position='${ifDefined(args.position)}'
    ?closed=${args.failed}
>${ifDefined(args.description)}</bl-toast>`

# Toast

Toast component displays an informational message for a while to users with additional features if desired.

### Basic Usage

* Set innerHTML as a description for Toast component.
* Use with 4 variants; **Info**, **Warning**, **Success** and **Danger**.
* Use with 4 positions; **top-right**, **top-left**, **bottom-right** and **bottom-left**.
* Use with 'close' argument to hide Toast component.

## Toast Variants

We have 4 variants for each toast: **Info**, **Warning**, **Success** and **Danger**.
